47 template <typename Tp = double, typename Tp1 = double>
48 void calcSingleSegmentPose(Tp1 L, Tp1 theta, Tp1 delta, Pose<Tp>& pose)
49 {
50  Eigen::Matrix<Tp, 3, 3> R_t1_2_tb =
51  rotByZ<Tp>(-PI / 2 + delta)*rotByY<Tp>(-PI / 2);
52  pose.R = R_t1_2_tb * rotByZ<Tp>(theta) * R_t1_2_tb.transpose();
53  if (abs(theta) < 1e-5) {
54  pose.t = { 0, 0, L };
55  }
56  else {
57  Tp rc = L / theta;
58  pose.t = rc * R_t1_2_tb *
59  Eigen::Vector<Tp, 3>(sin(theta), 1 - cos(theta), 0);
60  }
61 }
